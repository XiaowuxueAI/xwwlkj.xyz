四、实战：写一个能赚钱的自动化脚本

下面我们写一个真实可用的脚本——自动监控某商品价格，降价时发邮件通知。这类脚本在猪八戒上至少能卖200元。

4.1 价格监控脚本（完整代码）

import requests
import smtplib
import time
from email.mime.text import MIMEText

# ============ 配置区 ============
TARGET_URL = "https://item.jd.com/100012345678.html"  # 替换成你要监控的商品
TARGET_PRICE = 99.0  # 低于这个价格就通知
CHECK_INTERVAL = 3600  # 每小时检查一次（秒）

# 发件邮箱配置（推荐用QQ邮箱）
SMTP_SERVER = "smtp.qq.com"
SMTP_PORT = 587
SENDER_EMAIL = "your_email@qq.com"
SENDER_PASS = "your_auth_code"  # QQ邮箱授权码，不是密码
RECEIVER_EMAIL = "receiver@qq.com"
# ================================

def get_price():
    """获取商品当前价格（示例用京东，实际需根据页面结构调整）"""
    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) "
                      "AppleWebKit/537.36 (KHTML, like Gecko) "
                      "Chrome/120.0.0.0 Safari/537.36"
    }
    try:
        resp = requests.get(TARGET_URL, headers=headers, timeout=10)
        # 实际使用时需要解析页面提取价格
        # 这里用正则示例：re.search(r'"price":"([\d.]+)"', resp.text)
        # 为演示简化，返回模拟数据
        return 88.5
    except Exception as e:
        print(f"获取价格失败: {e}")
        return None

def send_alert(current_price):
    """发送降价提醒邮件"""
    subject = f"💰 降价提醒！当前价格：¥{current_price}"
    body = f"您监控的商品已降价至 ¥{current_price}，低于目标价 ¥{TARGET_PRICE}，建议尽快下手！\n\n商品链接：{TARGET_URL}"
    
    msg = MIMEText(body, "plain", "utf-8")
    msg["Subject"] = subject
    msg["From"] = SENDER_EMAIL
    msg["To"] = RECEIVER_EMAIL
    
    try:
        server = smtplib.SMTP(SMTP_SERVER, SMTP_PORT)
        server.starttls()
        server.login(SENDER_EMAIL, SENDER_PASS)
        server.sendmail(SENDER_EMAIL, [RECEIVER_EMAIL], msg.as_string())
        server.quit()
        print("✅ 降价邮件已发送！")
    except Exception as e:
        print(f"❌ 邮件发送失败: {e}")

def main():
    print("🚀 价格监控脚本已启动...")
    print(f"📌 监控商品: {TARGET_URL}")
    print(f"🎯 目标价格: ¥{TARGET_PRICE}")
    print(f"⏱️  检查间隔: {CHECK_INTERVAL}秒\n")
    
    while True:
        price = get_price()
        if price is not None:
            print(f"[{time.strftime('%Y-%m-%d %H:%M:%S')}] 当前价格: ¥{price}")
            if price <= TARGET_PRICE:
                send_alert(price)
                break  # 通知后退出，如需持续监控可去掉这行
        time.sleep(CHECK_INTERVAL)

if __name__ == "__main__":
    main()

4.2 脚本说明与改进方向

六、避坑指南：新手最容易踩的3个雷区

6.1 爬虫别越过法律红线

抓取公开数据一般是合法的，但要注意：不要爬取需要登录才能看的内容、不要高频请求拖垮对方服务器、不要出售涉及个人隐私的数据。遵守 robots.txt 协议，每次请求间隔1-3秒，做个有道德的爬虫手。

接单不是写完代码就完事。客户往往不懂技术，你需要给一份简单的使用说明（截图+文字），告诉对方怎么运行、怎么改配置。好评率高的秘诀就在这里。
