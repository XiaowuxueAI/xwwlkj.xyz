### 1. Excel批量合并工具
```python
# 把文件夹里所有Excel文件合并成一个
import pandas as pd
import os
from glob import glob

folder = './excel_files/'  # 改成你的文件夹路径
files = glob(folder + '*.xlsx')

dfs = []
for f in files:
    df = pd.read_excel(f)
    dfs.append(df)
    print(f'已读取: {os.path.basename(f)} → {len(df)}行')

result = pd.concat(dfs, ignore_index=True)
result.to_excel('合并结果.xlsx', index=False)
print(f'完成！共合并{len(files)}个文件，{len(result)}行数据')
```

### 2. 文件自动分类整理
```python
# 按文件类型自动分类到不同文件夹
import os
import shutil
from pathlib import Path

source = './downloads/'  # 要整理的文件夹
rules = {
    '图片': ['.jpg', '.png', '.gif', '.webp', '.bmp'],
    '文档': ['.pdf', '.docx', '.txt', '.xlsx', '.pptx'],
    '视频': ['.mp4', '.avi', '.mov', '.mkv'],
    '压缩包': ['.zip', '.rar', '.7z', '.tar'],
    '代码': ['.py', '.js', '.html', '.css', '.json'],
}

for file in os.listdir(source):
    filepath = os.path.join(source, file)
    if not os.path.isfile(filepath):
        continue
    ext = Path(file).suffix.lower()
    for folder, exts in rules.items():
        if ext in exts:
            target = os.path.join(source, folder)
            os.makedirs(target, exist_ok=True)
            shutil.move(filepath, os.path.join(target, file))
            print(f'{file} → {folder}/')
            break

print('整理完成！')
```

### 5. PDF批量处理
```python
# 提取PDF中的文字内容
import PyPDF2

def extract_pdf_text(filepath):
    with open(filepath, 'rb') as f:
        reader = PyPDF2.PdfReader(f)
        text = ''
        for page in reader.pages:
            text += page.extract_text() + '\n'
        return text

# 批量处理文件夹内所有PDF
import os
for file in os.listdir('./pdfs/'):
    if file.endswith('.pdf'):
        text = extract_pdf_text(f'./pdfs/{file}')
        with open(f'./pdfs/{file}.txt', 'w', encoding='utf-8') as f:
            f.write(text)
        print(f'已提取: {file}')
```

### 6. 定时发送邮件
```python
# 每天定时发送邮件报告
import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart
import schedule
import time

def send_report():
    sender = 'your@email.com'
    password = 'your_password'  # 建议使用应用专用密码
    receiver = 'boss@email.com'

    msg = MIMEMultipart()
    msg['Subject'] = f'每日数据报告 - {time.strftime("%Y-%m-%d")}'
    msg['From'] = sender
    msg['To'] = receiver

    # 这里放你的报告内容
    body = '今日数据汇总：\n- 新增用户：128人\n- 成交订单：35单\n- 总金额：¥5,230'
    msg.attach(MIMEText(body, 'plain', 'utf-8'))

    with smtplib.SMTP_SSL('smtp.qq.com', 465) as server:
        server.login(sender, password)
        server.send_message(msg)
    print(f'报告已发送！{time.strftime("%H:%M:%S")}')

# 每天早上9点发送
schedule.every().day.at('09:00').do(send_report)

while True:
    schedule.run_pending()
    time.sleep(60)
```

## API开发实战教程

### 什么是API？
API就是"外卖菜单"——你不需要知道厨房怎么做菜，只需要点菜（发请求），就能拿到菜（返回数据）。

### 快速上手：调用OpenAI API

```python
import openai

# 1. 设置API密钥（在 platform.openai.com 获取）
client = openai.OpenAI(api_key='sk-你的密钥')

# 2. 发送请求
response = client.chat.completions.create(
    model='gpt-3.5-turbo',
    messages=[
        {'role': 'user', 'content': '用一句话介绍人工智能'}
    ]
)

# 3. 获取结果
print(response.choices[0].message.content)
```

### 调用DeepSeek API（更便宜！）

```python
import requests

url = 'https://api.deepseek.com/v1/chat/completions'
headers = {
    'Authorization': 'Bearer 你的DeepSeek密钥',
    'Content-Type': 'application/json'
}
data = {
    'model': 'deepseek-chat',
    'messages': [{'role': 'user', 'content': '你好'}]
}

response = requests.post(url, json=data, headers=headers)
print(response.json()['choices'][0]['message']['content'])
```

### 搭建你自己的API服务

```python
# 用Flask搭建一个简单的API
from flask import Flask, request, jsonify

app = Flask(__name__)

# 你的数据（这里用字典模拟）
database = {
    'products': [
        {'id': 1, 'name': 'AI提示词包', 'price': 1},
        {'id': 2, 'name': '文案模板', 'price': 9.9},
    ]
}

# 获取产品列表
@app.route('/api/products', methods=['GET'])
def get_products():
    return jsonify(database['products'])

# 获取单个产品
@app.route('/api/products/<int:id>', methods=['GET'])
def get_product(id):
    product = next((p for p in database['products'] if p['id'] == id), None)
    if product:
        return jsonify(product)
    return jsonify({'error': 'Not found'}), 404

# 新增产品
@app.route('/api/products', methods=['POST'])
def add_product():
    data = request.json
    data['id'] = len(database['products']) + 1
    database['products'].append(data)
    return jsonify(data), 201

if __name__ == '__main__':
    app.run(debug=True, port=5000)
```

运行后，访问 http://localhost:5000/api/products 即可看到你的API！
